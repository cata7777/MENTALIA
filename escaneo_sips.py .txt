    for file in os.listdir(path):

            try:
                with zipfile.ZipFile(full_path, 'r') as z:
                    print(f"ğŸ—‚ï¸ {file}: {len(z.namelist())} archivos internos")
            except zipfile.BadZipFile:
                print(f"âš ï¸ Error: ZIP corrupto â†’ {file}")
